Vectors.h

Implementation
+---------------------------------------------------------------+
Provides miscellaneous vector functions

Also includes:
<vector>
<cmath>


functions:

size_t find<typename T>(vector<T>, T element)
    return index of element in vector of type T
    returns string::npos (-1) if not found

void sortVec(vector<string>&)
    sorts vector of strings alphabetically;

[double/int/size_t] sum(vector<[double/int/size_t]>)
    returns sum of elements in vector

string toString(vector<[string/int]>, sep="\n")
    return string of elements in vector, separated by sep

string toString(vector<vector<char> >, sep="")
    converts 2d vector of characters to a string

vector<string> charToStrVec(vector<char>)
    converts a vector of characters to a vector of strings

vector<string> toVecOfStr(vector<[double/int/size_t]>)
    converts vector of numbers to vector of strings
